158  void check_vault_beginning(double height, const cv::Rect2d &person, std::size_t frame_no) {
159  std::vector<cv::Point2d> offsets;
160  if (dir == right) offsets = left_offsets;
161  else if (dir == left) offsets = right_offsets;
162 
163  if ((!_vault_began) && (offsets.size() > vault_check_frames)) {
164  double size = (double)person.height / height;
165  double runup_mean_delta = count_mean_delta(offsets.begin(), offsets.end() - vault_check_frames).y;
166  double vault_mean_delta = count_mean_delta(offsets.end() - vault_check_frames, offsets.end()).y;
167 
168  if ((vault_mean_delta - runup_mean_delta) * size / height < vault_threshold)
169  _vault_began = frame_no;
