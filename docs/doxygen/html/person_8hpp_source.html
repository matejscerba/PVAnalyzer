286  void extract_points(cv::Mat &output, std::size_t frame_no) {
287  points.push_back(frame_body(npoints));
288 
289  int h = output.size[2];
290  int w = output.size[3];
291 
292  // Scale by `scaling_factor` if last rectangle was scaled.
293  double factor = scaling_performed ? scale_factor : 1;
294  double sx = factor * width(frame_no) / (double)w;
295  double sy = factor * height(frame_no) / (double)h;
296 
297  // Get points from output.
298  for (int n = 0; n < npoints; n++) {
299  cv::Mat probMat(h, w, CV_32F, output.ptr(0, n));
300 
301  // Get point in output with maximum probability of "being point `n`".
302  frame_part p = std::nullopt;
303  cv::Point max;
304  double prob;
305  cv::minMaxLoc(probMat, 0, &prob, 0, &max);
306 
307  // Check point probability against a threshold
308  if (prob > detection_threshold) {
309  p = max;
310  p->x *= sx; p->y *= sy; // Scale point so it fits original frame.
311 
312  // Move point `p` so it is in correct position in frame.
313  p = corners[frame_no - first_frame_no][corner::tl]
314  + (1.0 - scale_factor) * 0.5 * (corners[frame_no - first_frame_no][corner::br] - corners[frame_no - first_frame_no][corner::tl])
315  + p->x * (corners[frame_no - first_frame_no][corner::tr] - corners[frame_no - first_frame_no][corner::tl]) / width(frame_no)
316  + p->y * (corners[frame_no - first_frame_no][corner::bl] - corners[frame_no - first_frame_no][corner::tl]) / height(frame_no);
317  }
318 
319  points[frame_no - first_frame_no][n] = p;
320  }
321  }

406  cv::Rect scale(const cv::Rect &rect, const cv::Mat &frame) {
407  cv::Point center(rect.x + rect.width / 2, rect.y + rect.height / 2);
408  cv::Point diag = center - rect.tl();
409  cv::Point tl = center - scale_factor * diag;
410  cv::Point br = center + scale_factor * diag;
