136  void extract_points(cv::Mat &output, std::size_t frame_no) {
137  points.push_back(std::vector<cv::Point2d>(npoints));
138 
139  int h = output.size[2];
140  int w = output.size[3];
141 
142  // Scale by `scaling_factor` if last rectangle was scaled.
143  double factor = scaling_performed ? scale_factor : 1;
144  double sx = factor * width(frame_no) / (double)w;
145  double sy = factor * height(frame_no) / (double)h;
146 
147  // Get points from output.
148  for (int n = 0; n < npoints; n++) {
149  cv::Mat probMat(h, w, CV_32F, output.ptr(0, n));
150 
151  // Get point in output with maximum probability of "being point `n`".
152  cv::Point2d p(-1, -1);
153  cv::Point max;
154  double prob;
155  cv::minMaxLoc(probMat, 0, &prob, 0, &max);
156 
157  // Check point probability against a threshold
158  if (prob > probThreshold) {
159  p = max;
160  p.x *= sx; p.y *= sy; // Scale point so it fits original frame.
161 
162  // Move point `p` so it is in correct position in frame.
163  p = corners[frame_no - first_frame_no][corner::tl]
164  + (1.0 - scale_factor) * 0.5 * (corners[frame_no - first_frame_no][corner::br] - corners[frame_no - first_frame_no][corner::tl])
165  + p.x * (corners[frame_no - first_frame_no][corner::tr] - corners[frame_no - first_frame_no][corner::tl]) / width(frame_no)
166  + p.y * (corners[frame_no - first_frame_no][corner::bl] - corners[frame_no - first_frame_no][corner::tl]) / height(frame_no);
167  }
168 
169  points[frame_no - first_frame_no][n] = p;
170  }
171  }

262  cv::Rect scale(const cv::Rect &rect, const cv::Mat &frame) {
263  cv::Point center(rect.x + rect.width / 2, rect.y + rect.height / 2);
264  cv::Point diag = center - rect.tl();
265  cv::Point tl = center - scale_factor * diag;
266  cv::Point br = center + scale_factor * diag;
267 
268  if (tl.x >= 0 && tl.y >= 0 && br.x <= frame.cols && br.y <= frame.rows) {
269  // Make sure it fits inside frame.
270  scaling_performed = true; // Remember if scaling was performed.
271  return cv::Rect(tl, br);
272  } else {
273  scaling_performed = false;
274  return rect;

407  void draw(cv::Mat &frame, std::size_t frame_no) const {
408  // Check whether person exists in given frame.
409  if (frame_no < first_frame_no) {
410  std::cout << "Unable to draw frame no. " << frame_no << std::endl;
411  return;
412  }
413  std::size_t idx = frame_no - first_frame_no;
414 
415  if (corners.size() > idx) {
416  // Rectangle which is tracked.
417  cv::Point2d tl = corners[idx][corner::tl];
418  cv::Point2d tr = corners[idx][corner::tr];
419  cv::Point2d bl = corners[idx][corner::bl];
420  cv::Point2d br = corners[idx][corner::br];
421 
422  cv::Scalar color(0, 0, 255);
423  if (move_analyzer.vault_began(frame_no))
424  color = cv::Scalar(0, 255, 0);
425 
426  cv::line(frame, tl, tr, color, 1);
427  cv::line(frame, tr, br, color, 1);
428  cv::line(frame, br, bl, color, 1);
429  cv::line(frame, bl, tl, color, 1);
430 
431  // Scaled rectangle.
432  tl = corners[idx][corner::tl]
433  + (1.0 - scale_factor) * 0.5 * (corners[idx][corner::br] - corners[idx][corner::tl]);
434  tr = corners[idx][corner::tr]
435  + (1.0 - scale_factor) * 0.5 * (corners[idx][corner::bl] - corners[idx][corner::tr]);
436  bl = corners[idx][corner::bl]
437  - (1.0 - scale_factor) * 0.5 * (corners[idx][corner::bl] - corners[idx][corner::tr]);
438  br = corners[idx][corner::br]
439  - (1.0 - scale_factor) * 0.5 * (corners[idx][corner::br] - corners[idx][corner::tl]);
440 
441  color = cv::Scalar(0, 0, 127);
442  if (move_analyzer.vault_began(frame_no))
443  color = cv::Scalar(0, 127, 0);
444 
445  cv::line(frame, tl, tr, color, 1);
446  cv::line(frame, tr, br, color, 1);
447  cv::line(frame, br, bl, color, 1);
448  cv::line(frame, bl, tl, color, 1);
449  }
450 
451  // Body parts if they were detected.
452  if (points.size() > idx) {
453  for (int n = 0; n < npairs; n++) {
454  cv::Point2d a = points[idx][pairs[n][0]];
455  cv::Point2d b = points[idx][pairs[n][1]];
456 
457  // Check if points `a` and `b` are valid.
458  if (a.x <= 0 || a.y <= 0 || b.x <= 0 || b.y <= 0)
459  continue;
460 
461  // Draw points representing joints and connect them with lines.
462  cv::line(frame, a, b, cv::Scalar(0, 255, 255), 2);
463  cv::circle(frame, a, 2, cv::Scalar(0, 0, 255), -1);
464  cv::circle(frame, b, 2, cv::Scalar(0, 0, 255), -1);
465  }
466  }
467 
468  // Movement analyzer.
469  move_analyzer.draw(frame);
470  }
