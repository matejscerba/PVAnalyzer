208  cv::Rect get_tracker_bbox(const cv::Mat &frame, const frame_body &body, const cv::Rect &bbox) noexcept {
209  if ((body[body_part::l_hip] || body[body_part::r_hip]) && body[body_part::head]) {
210  std::vector<cv::Point2d> torso_corners;
211  cv::Point2d head = *body[body_part::head];
212  cv::Point2d hip;
213  if (body[body_part::l_hip] && body[body_part::r_hip])
214  hip = (*body[body_part::l_hip] + *body[body_part::r_hip]) / 2;
215  else if (body[body_part::l_hip])
216  hip = *body[body_part::l_hip];
217  else
218  hip = *body[body_part::r_hip];
219  cv::Point2d torso = hip - head;
220  cv::Point2d perp(torso.y, - torso.x);
221  perp *= 1.0 / 4.0;
222  torso_corners.push_back(head + perp);
223  torso_corners.push_back(head - perp);
224  torso_corners.push_back(hip + perp);
225  torso_corners.push_back(hip - perp);
226  double top = torso_corners.front().y;
227  double right = torso_corners.front().x;
228  double bottom = torso_corners.front().y;
229  double left = torso_corners.front().x;
230  for (const auto &c : torso_corners) {
231  top = std::min(top, c.y);
232  right = std::max(right, c.x);
233  bottom = std::max(bottom, c.y);
234  left = std::min(left, c.x);
235  }
236  cv::Rect new_bbox(left, top, right - left, bottom - top);
237  return new_bbox;
238  }
