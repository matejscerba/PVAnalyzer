155  cv::Rect update_rect(const cv::Mat &frame, const cv::Rect &background, const std::vector<cv::Rect> &bboxes) noexcept {
156  cv::Rect bg = background;
157  if (!backgrounds.size() || !backgrounds.back() || bg.x <= 0 || bg.y <= 0 ||
158  bg.x + bg.width >= (double)frame.cols || area(bg & rect(bboxes)) > 0 ||

183  backgrounds.push_back(new_background);

186  // If background was not changed in `update_rect()`, background's offset remains the same.
187  cv::Point2d last_offset = *background_offsets.back();
188  cv::Point2d offset_change = get_center(new_background) - get_center(background);
189  background_offsets.push_back(last_offset + offset_change);
190  } else {

192  }
