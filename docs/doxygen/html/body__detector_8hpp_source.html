334  std::vector<cv::Rect> nms(const std::vector<cv::Rect> &rects) const noexcept {
335  std::vector<cv::Rect> res;
336  std::vector<cv::Rect> sorted = rects;
337  std::sort(sorted.begin(), sorted.end(), [](const auto &r, const auto &s){ return area(r) < area(s); });
338  while (sorted.size()) {
339  res.push_back(sorted.back());
340  sorted.erase(std::remove_if(sorted.begin(), sorted.end(), [&res](const auto &r){
341  cv::Rect intersection = res.back() & r;
342  return area(intersection) / area(r) >= NMS_THRESHOLD;
343  }));
344  }
345  return res;
346  }

358  void find(const cv::Mat &frame, std::size_t frame_no) noexcept {
