419  virtual void compute(const video_body &points) noexcept {
420  values.clear();
421  step_frames.clear();
422  std::greater<double> low;
423  std::vector<std::size_t> lows = get_frame_numbers(points.begin(), points.end(), low);
424  std::less<double> high;
425  std::vector<std::size_t> highs = get_frame_numbers(points.begin(), points.end(), high);
426  double center = 0;
427  for (std::size_t i = 0; i < std::min(lows.size(), highs.size()); ++i) {
428  frame_body l_body = points[lows[i]];
429  frame_body h_body = points[highs[i]];
430  double l = *get_height(l_body[body_part::l_ankle], l_body[body_part::r_ankle], low);
431  double h = *get_height(h_body[body_part::l_ankle], h_body[body_part::r_ankle], high);
432  if (i > 0) {
433  if (high(l, center)) continue; // Low point is above center of previous points.
434  if (low(h, center)) continue; // High point is below center of previous points.
435  }
436  center = (l + h) / 2.0;
437  step_frames.push_back(lows[i]);
438  }
439  for (std::size_t i = 1; i < step_frames.size(); ++i) {
440  values.push_back((step_frames[i] - step_frames[i - 1]) / fps);
441  }
442  }

506  std::vector<std::size_t> get_frame_numbers( std::vector<frame_body>::const_iterator begin,
507  std::vector<frame_body>::const_iterator end,
508  std::function<bool (double, double)> compare) noexcept {
509  std::vector<std::size_t> res;
510  std::optional<double> last_height = std::nullopt;
511  std::size_t index = 0;
512  bool correct_diff = false;
513  for (; begin != end; ++begin) {
514  std::optional<double> height = get_height((*begin)[body_part::l_ankle], (*begin)[body_part::r_ankle], compare);
515  if (height && last_height) {
516  // Current and last value is valid.
517  if (correct_diff && !compare(*height, *last_height) && std::abs(*height - *last_height) > 1) {
518  // Value was changing in the right direction, it stopped changing and is changing in the wrong direction.
519  correct_diff = false;
520  res.push_back(index - 1);
521  }
522  if (compare(*height, *last_height)) {
523  correct_diff = true;
524  }
525  }
526  last_height = height;
527  ++index;
528  }
529  return res;
530  }
